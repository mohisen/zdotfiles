abbrfile = "myfile.txt" 
 sep = '=' 
 folder = engine.get_folder("My Phrases") 
 
 f = open(abbrfile, 'r') 
 for line in f: 
 abbr, contents = line.split(sep) 
 if len(contents) > 20: 
 title = contents[0:17] + "..." 
 else: 
 title = contents 
 
 engine.create_abbreviation(folder, title, abbr, contents)
